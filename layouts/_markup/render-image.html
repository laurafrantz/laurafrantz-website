{{- /* Image render hook. For page-bundle raster images: with responsive WebP for browsers and a JPEG fallback. The JPEG fallback keeps images working in RSS readers, email, and Substack (which don't reliably render WebP). URLs are absolutized in the RSS template (layouts/_default/rss.xml). For static paths / external URLs / non-raster: fall back to a plain . Mirrors PaperMod's resolution logic (PageInner.Resources / global resources). */ -}} {{- $u := urls.Parse .Destination -}} {{- $src := $u.String -}} {{- $res := false -}} {{- if not $u.IsAbs -}} {{- $path := strings.TrimPrefix "./" $u.Path -}} {{- with or (.PageInner.Resources.Get $path) (resources.Get $path) -}} {{- $res = . -}} {{- $src = .RelPermalink -}} {{- end -}} {{- end -}} {{- $raster := slice "jpeg" "png" "tiff" "bmp" "webp" -}} {{- if and $res (in $raster $res.MediaType.SubType) -}} {{- $ow := $res.Width -}} {{- /* candidate widths, capped at the original; add the original itself if it falls below the top breakpoint so the sharpest version is offered too */ -}} {{- $ews := slice -}} {{- range $w := (slice 480 720 1080 1440) -}} {{- if le $w $ow -}}{{- $ews = $ews | append $w -}}{{- end -}} {{- end -}} {{- if and (lt $ow 1440) (not (in $ews $ow)) -}}{{- $ews = $ews | append $ow -}}{{- end -}} {{- if not $ews -}}{{- $ews = slice $ow -}}{{- end -}} {{- $set := slice -}} {{- range $w := $ews -}} {{- $d := $res.Resize (printf "%dx webp q82" $w) -}} {{- $set = $set | append (printf "%s %dw" $d.RelPermalink $w) -}} {{- end -}} {{- /* JPEG fallback for the (feed/email safe), capped at 1080px */ -}} {{- $jw := cond (gt $ow 1080) 1080 $ow -}} {{- $jpg := $res.Resize (printf "%dx jpg q82" $jw) -}} {{- $title := cond (ne (string .Title) "") (printf " title=%q" (.Title | transform.HTMLEscape)) "" -}} {{- else -}} {{- /* Not a resolvable resource (e.g. a not-yet-added placeholder or a static path). Make bundle-relative paths root-relative so they resolve on-site and can be absolutized in the RSS feed. */ -}} {{- if and (not $u.IsAbs) (not (strings.HasPrefix $src "/")) -}} {{- $src = printf "%s%s" .PageInner.RelPermalink $src -}} {{- end -}} {{- $attrs := merge .Attributes (dict "alt" .Text "src" $src "title" (.Title | transform.HTMLEscape) "loading" "lazy" "decoding" "async") -}} {{- end -}} {{- /**/ -}}
